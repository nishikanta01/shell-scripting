if statement:-

if [ condition ];
then
     statement
fi

2. if-else statement:-

if [ condition ];
then
     statement 1
else
     statement 2
fi

3. if-elif statement:-

if [ condition 1 ];
then
    statement 1
elif [condition 2 ];
then
     statement 2
else
     statement 3
fi

4.for loop:-

for var in list;
do
   statement
done

5.while loop:-

while [ condition ];
do
   statement
done

6.until loop:-

until [ ccondition ];
do
   statement
done

7.case statement:-

case expression in
   case1) statement ;;
   case2) statement ;;
   case3) statement ;;
   
   *) statement ;;
esac
